TABLE WITHOUT ID file.link AS 개념, file.folder AS 분류, status AS 상태
FROM "01_운영체제" OR "02_네트워크" OR "03_데이터베이스" OR "04_자료구조" OR "05_알고리즘" OR "06_디자인패턴" OR "07_컴퓨터구조" OR "08_소프트웨어공학"

SORT status ASC, file.folder ASC
